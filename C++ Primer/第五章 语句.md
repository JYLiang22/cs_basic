## throw表达式
之前写过的关于Sales_item对象相加的程序，是像下面这样的：    
```cpp
Sales_item item1, item2;
cin >> item1 >> item2;
// 首先检查item1和item2是否表示同一种书籍
if(item1.isbn() == item2.isbn()){
  cout << item1 + item2 << endl;
  return 0;
}
else{
  cerr << "Data must refer to same ISBN" << endl;
}
```

使用throw书写如下：   
```cpp
if(item1.isbn() != item2.isbn())
  throw runtime_error("Data must refer to same ISBN");
cout << item1 + item2 << endl;
```

5.24 使用throw表达式检测异常：  
```cpp
#include<iostream>
#include<stdexcept>
using namespace std;

int main(){

    cout << "input two nums for div: " << endl;
    int val1, val2;
    cin >> val1 >> val2;
    if(val2 == 0){
        throw runtime_error("number of divisor can't be 0!");

    cout << "reselt is " << val1 / val2 << endl;

    return 0;
}
```

5.25 使用try语句块处理异常：   
```cpp
#include<iostream>
#include<stdexcept>
using namespace std;

int main(){

    cout << "input two nums for div: " << endl;
    int val1, val2;
    while(cin >> val1 >> val2){
        try{
            if(val2 == 0){
                throw runtime_error("number of divisor can't be 0!");
            }
            cout << "result is " << val1 / val2 << endl;
            cout << "want to exit?(y or n)" << endl;
            char exit;
            cin >> exit;
            if(exit == 'y' || exit == 'Y'){
                break;
            }
        }catch(runtime_error err){
            cout << err.what() << endl;
            cout << "need to continue?(y or n)" << endl;
            char ch;
            cin  >> ch;
            if(ch != 'y' && ch != 'Y'){
                break;
            }
            cout << "input two nums for div: " << endl;
        }
    }

    return 0;
}
```
